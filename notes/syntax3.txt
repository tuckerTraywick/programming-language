pub package animals
from std.io import printLine

pub struct Animal {
    pub name string
    pub age uint
    pub makeNoise(&this)
}

pub struct Dog {
    embed Animal
    pub tag string
}

pub struct Cat {
    embed Animal
    pub breed string
}

pub fun makeNoise(dog &Dog) {
    printLine("woof")
}

pub fun makeNoise(cat &Cat) {
    printLine("meow")
}

pub package std.collections

pub struct List[T] {
    first mut optional owned &Node[T]
    last mut optional &Node[T]
    length mut usize
}

struct Node[T] {
    next mut optional owned &Node[T]
    previous mut optional &Node[T]
    value mut T
}
