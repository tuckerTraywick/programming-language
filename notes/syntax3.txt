pub package physics
import std.operations

pub struct Entity
    pub position, velocity, acceleration Vector
union
    struct Circle
        pub radius float64
    end

    struct Mesh
        pub vertices owned []Vector
    end
end

pub struct Vector
    default std.operations.Arithmetic, .Equality
    std.operations.To string
    pub x, y, z float64
end




pub package std.collections
import std.operations

pub struct LinkedList[T]
    std.operations.TryArithmetic, .TryBitwise, .TryBoolean
    std.operations.TryGetElement, .TrySetElement
    std.operations.To string

    first mut maybe owned &Node T
    last mut maybe &Node T
end

for LinkedList[T] impl std.operations.TryArithmetic, .TryBitwise, .TryBoolean


struct Node[T]
    data T
    next mut maybe owned &this
    previous mut maybe &this
end




package lexer
import std.File, .FileError

struct Token
    type TokenType
    index, row, column, length usize
end

union TokenType
    default invalid
    identifier, number, character, string_
    package_, import_, ...
end

func lexFile(path string) union(owned []Token, FileError)

func lexFile(file File) union(owned []Token, FileError)
