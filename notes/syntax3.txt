    pub position, velocity, acceleration Vector
cases
    struct Circle
        pub radius float64
    end

        pub vertices owned []Vector
    end

pub struct Vector
    from std.operations embed default Arithmetic, Equality
    from std.operations embed To string

end




pub package std.collections
import std.operations

pub struct LinkedList[T]
    from std.operations embed TryArithmetic, TryBitwise, TryBoolean
    from std.operations embed TryGetElement, TrySetElement
    from std.operations embed To string

    first mut maybe owned &Node T
    last mut maybe &Node T
end

struct Node[T]
    data T
    next mut maybe owned &this
    previous mut maybe &this
end
