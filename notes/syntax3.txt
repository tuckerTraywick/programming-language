pub package animals
from std.io import printLine

pub struct Animal {
    pub name string
    pub age uint
    pub makeNoise(&this)
}

pub struct Dog {
    embed Animal
    pub tag string
}

pub struct Cat {
    embed Animal
    pub breed string
}

pub fun makeNoise(dog &Dog) {
    printLine("woof")
}

pub fun makeNoise(cat &Cat) {
    printLine("meow")
}

pub package std.compiler
from std.io import File, FileError

pub struct TokenType {
    pub value uint = auto
    pub name string = auto
} cases {
    default invalid
    number
    character
    string
    identifier
    newline
    ...
}

pub struct Token {
    pub type TokenType
    pub text string
    pub index, row, column usize
}

pub struct LexingError {
    embed Error
    pub message string
    pub token &Token
}

pub struct LexingResult {
    pub tokens optional owned []mut Token
    pub errors optional owned []LexingError
}

pub struct NodeType {
    pub value uint = auto
    pub name string = auto
} cases {
    token
    program
    packageStatement
    ...
}

pub struct Node {
    pub type NodeType
    pub tokens []Token
    pub next, parent, child &Node
}

pub struct ParsingError {
    embed Error
    pub message string
    pub tokens []Token
    pub index usize
}

pub struct ParsingResult {
    pub nodes optional owned []mut Node
    pub errors optional owned []&ParsingError
}

pub fun lexFile(path string) LexingResult or FileError

pub fun lexFile(file File) LexingResult or FileError

pub fun lex(text string) LexingResult

pub fun parseFile(path string) ParsingResult or FileError

pub fun parseFile(file File) ParsingResult or FileError

pub fun parse(text string) (LexingResult, ParsingResult)

pub fun parse(tokens []Token) ParsingResult




pub package std.collections

pub struct List[T] {
    first mut optional owned &Node[T]
    last mut optional &Node[T]
    length mut usize
}

struct Node[T] {
    next mut optional owned &Node[T]
    previous mut optional &Node[T]
    value mut T
}

pub fun .length(list &List) = auto

@discard=Error
pub fun [][T](list &List[T]) T or IndexOutOfBoundsError {}

@discard=Error
pub fun []=[T](list &List[T], T) optional IndexOutOfBoundsError {}

@discard=Error
pub fun append[T](list &List[T], value &T) optional AllocationError {}
