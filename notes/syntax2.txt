pub namespace animals;
using std.io.printLine;

pub trait Animal;
pub .name(*Animal);
pub .age(*Animal) uint32;
pub makeNoise(*Animal);

pub struct Dog {
	Animal;
	pub tag []char8;
}

pub struct Cat {
	Animal;
	pub breed []char8;
}

pub makeNoise(dog *Dog) {
	printLine("woof");
}

pub makeNoise(cat *Cat) {
	printLine("meow");
}

main() {
	pet mut *Animal = &Dog();
	makeNoise(pet);
	pet = &Cat();
	pet->makeNoise();
}









pub namespace animals;
using std.io.{printLine};

pub trait Noisy;
pub []char8 .name(*Noisy noisy);
pub uint32 .age(*Noisy noisy);
pub void makeNoise(*Noisy noisy);

pub struct Dog {
	Noisy;
	pub []char8 tag;
}

pub struct Cat {
	Noisy;
	pub []char8 breed;
}

pub void makeNoise(*Dog dog) {
	printLine("woof");
}

pub void makeNoise(*Cat cat) {
	printLine("meow");
}

void main() {
	mut *Noisy pet = &Dog();
	makeNoise(pet);
	pet = &Cat();
	pet->makeNoise();
}




pub namespace physics;
using std.operators;

pub struct Entity {
	Vector position, velocity, acceleration;
} cases {
	Circle, Mesh,
}
pub void update(*mut Entity entity);

pub struct Circle {
	Entity;
	pub float64 radius;
}

pub struct Mesh {
	Entity;
	pub owned []Vector vertices;
}

pub struct Vector {
	std.operators.{To<[]char8>, DefaultArithmetic};
	float64 x, y, z;
}

pub void update(*mut Circle circle) {}

pub void update(*mut Mesh mesh) {}

pub []char8 to<[]char8>(*Vector vector) {}
