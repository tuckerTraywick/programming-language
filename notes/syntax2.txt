pub module animals;
import std.io.printLine;

pub trait Animal;
pub string .name(*Animal animal);
pub uint32 .age(*Animal animal);
pub void makeNoise(*Animal animal);

pub struct Dog {
	Animal;
	pub string tag;
}

pub struct Cat {
	Animal;
	pub string breed;
}

pub void makeNoise(*Dog dog) {
	printLine("woof");
}

pub void makeNoise(*Cat cat) {
	printLine("meow");
}

void main() {
	mut *Animal pet = &Dog{};
	makeNoise(pet);
	pet = &Cat{};
	pet->makeNoise();
}




pub module physics;
import std.operators;

pub closed struct Entity {
	pub Vector position, velocity, acceleration;
}
pub void collide(*mut Entity a, b);

pub struct Circle {
	Entity;
	pub float64 radius;
}

pub struct Mesh {
	Entity;
	pub owned []Vector vertices;
}

pub struct Vector {
	std.operators.To<string>, .DefaultEquality, .DefaultArithmetic;
	pub float64 x, y, z;
}

pub void collide(*mut Circle a, b) {}

pub void collide(*mut Mesh a, b) {}

pub void collide(*mut Circle a, *mut Mesh b) {}

pub void collide(*mut Mesh a, *mut Circle b) {}
