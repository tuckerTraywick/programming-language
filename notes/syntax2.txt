package animals

struct Animal {
    name string
    age uint
}

struct Dog {
    extend Animal
    tag string
}

struct Cat {
    extend Animal
    breed string
}

fun makeNoise(animal &Animal)

fun makeNoise(dog &Dog) => printLine("woof")

fun makeNoise(cat &Cat) => printLine("meow")

fun main() {
    var animal &Animal = &Dog()
    makeNoise(animal)
    animal = &Cat()
    animal->makeNoise()
}

    color [3]uint8
}

}

    extend Shape
    length, width ufloat64
