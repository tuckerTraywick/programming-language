public package animals
import std.io: printLine

public struct Animal {
    public name string
    public age uint
    public makeNoise()
}

public struct Dog {
    Animal
    public tag string
}

struct Cat {
    Animal
    public breed string
}

public func makeNoise(dog &Dog) {
    printLine("woof")
}

public func makeNoise(cat &Cat) {
    printLine("meow")
}
