import std.operations
import std.math

pub struct Entity {
    pub position, velocity acceleration Vector
} cases {
    struct Circle {
        pub radius float64
    }

    struct Mesh {
        pub vertices owned []Vector
    }
}

pub struct Vector {
    from std.operations embed To string
    from std.operations embed default Arithmetic, Bitwise, Equality
    pub x, y, z float64
}

pub func collide(a, b &mut Entity) {
    match a, b {
        case .Circle, .Circle:
        case .Mesh, .Mesh:
        case .Circle, .Mesh:
        case .Mesh, .Circle:
    }
}

pub func to[string](v &Vector) {
    return f"({", ".split(v.(x, y, z))})"
}

pub func length(v &Vector) {
    return math.sqrt(v.x*v.x + v.y*v.y + v.z*v.z)
}

func run(editor &Editor) {
    tui.begin()

    tui.clear()
    drawFileExplorer(editor)
    tui.end()

func drawFileExplorer(editor &Editor) {
    if fullscreen {
        tui.printLine("File Explorer")
        drawDirectory(editor)
    }
}

func drawDirectory(editor &Editor, depth usize=0) {
    assert editor.mode is .FileExplorer
    var directory = editor.mode.currentDirectory
    assert io.isPath(directory)
    if editor.mode.currentSelection == index {
        tui.enableStyle(.invert)
    }
    tui.printLine("\t"*depth + subdirectory->io.tail())
    tui.disableStyle(.invert)
    for var subdirectory in io.subdirectories(directory) {
        drawDirectory(editor, depth + 1)
    }
}

func main() {
    Editor().run()
}
