public package animals
from std.io import printLine

public struct Animal {
    public name string
    public age uint
    public makeNoise(this)
}

public struct Dog {
    Animal
    public tag string
}

public struct Cat {
    Animal
    public breed string
}

public func makeNoise(dog &Cat) {
    printLine("woof")
}

public func makeNoise(cat &Cat) {
    printLine("meow")
}

public func main() {
    var pet &Animal = &Dog()
    pet.makeNoise()
    pet = &Cat()
    pet.makeNoise()
}
