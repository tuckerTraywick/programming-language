# No `implement` keyword because you can overload methods in subtypes, which makes it a moot point
# whether one type extends or implements another. The only difference is whether you can construct
# an instance of it.

# Types can be abstract or closed. Abstract types can have abstract functions, closed types cannot
# be extended and can have abstract functions. Closed types are also abstract.

type UserId = uint32


type Direction
    extends uint32
    case n, s, e, w
end


closed type Color
    var r, g, b uint8
end

type Rgb
    extends default Color
end

type ColorCases
    extends Color
    case red = (255,0,0), green = (0,255,0), blue = (0,0,255)
end

pub package physics
import std.operations
import std.math

pub closed type Entity
	pub var pos, vel, acc Vector
end

pub type Circle
	extends Entity
	pub var radius float64
end

pub type Mesh
	extends Entity
	pub var vertices owned []Vector
end

pub type Vector
	from std.operations import *
	extends Equality, Arithmetic
	pub var x, y, z float64
end

pub fun collide(a, b &mut Entity)
	match *a, *b
		case Circle, Circle
			pass
		case Mesh, Mesh
			pass
		case Circle, Mesh
			pass
		case Mesh, Circle
			pass
	end
end

pub fun .length(v &Vector) float64
	return math.sqrt(v.x*v.x + v.y*v.y)
end




pub package lexer
from std.io import File->*, FileError->*

pub type TokenType
	extends uint32
	default case invalid
	case identifier, number, string, ...

pub type Token

	pub var text string
	pub var index usize
end

pub fun lex(file File) any[owned []mut Token, FileError]

pub fun lex(text string) owned []mut Token
