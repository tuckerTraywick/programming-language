pub namesepace example;
import std.io;

struct AnimalMethods {
    usize offset;
    void(void*) makeNoise;
    void(void*) walk;
}

var DogMethods = AnimalMethods{
    offsetof(Dog.Animal), DogMakeNoise, DogWalk
};

var CatMethods = AnimalMethods{
    offsetof(Cat.Animal), CatMakeNoise, CatWalk
};

pub struct Animal {
    AnimalMethods* methods;
    pub string name;
    pub uint32 age;
}

pub struct Dog {
    Animal;
    pub string tag;
}

pub struct Cat {
    Animal;
    pub string breed;
}

pub void AnimalMakeNoise(Animal* animal) {
    animal.methods.makeNoise(animal as(byte*) - animal.methods.offset);
}

pub void AnimalWalk(Animal* animal) {
    animal.methods.walk(animal as(byte*) - animal.methods.offset);
}

pub void DogCreate() {
    return Dog{methods=DogMethods};
}

pub void DogMakeNoise(Dog* dog) {
    std.io.printLine("woof");
}

pub void CatCreate() {
    return Cat{methods=CatMethods};
}

pub void CatMakeNoise(Cat* cat) {
    std.io.printLine("meow");
}

pub void main() {
    Animal* pet = &DogCreate();
    AnimalMakeNoise(pet);
    pet = &CatCreate();
    pet->AnimalMakeNoise();
}
