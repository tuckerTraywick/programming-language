pub package animals
from std.io import printLine

pub trait Noisy {
    pub makeNoise(this &Animal)
}

pub struct Animal {
    pub name string
    pub age uint32
}

pub struct Dog {
    extend Animal
    implement Noisy
    pub tag string
}

pub struct Cat {
    extend Animal
    implement Noisy
    pub breed string
}

pub func makeNoise(dog &Dog) {
    printLine("woof")
}

pub func makeNoise(dog &Dog) {
    printLine("woof")
}

pub func makeNoise(cat &Cat) {
    printLine("meow")
}


pub package moreAnimals
from animals import *

pub trait Walkable {
    abstract Dog, Cat
    pub walk(this &Pet)
}

pub struct Pet {
    extend Animal
    abstract Dog, Cat
}

pub struct Hamster {
    extend Pet
    implement Walkable
}




pub package std.collections
import std.operations

pub struct List[T implements Sized] {
    from std.operations import *
    implement default Equality, TryComparison, TryArithmetic, TryBitwise, TryLogical
    implement default To[bool], To[string], To[strong []T], From[[]T]
    length usize
    elements strong []mut T
}

pub func append[T](list &mut List[T], value &T)




pub package physics
import std.math

pub closed struct Entity {
    pub pos, vel, acc Vector
}

pub struct Circle {
    extend Entity
    pub radius float64
}

pub struct Mesh {
    extend Entity
    pub vertices []Vector
}

pub struct Vector {
    from std.operations import *
    extend Equality, Arithemtic, To[string]
    pub x, y, z float64
}

pub func length(v &Vector) float64

pub func collide(a, b &mut Entity)
