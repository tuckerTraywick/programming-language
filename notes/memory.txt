package std.collections;
public const ufloat arrayGrowthRate = 1.5;
// You can append to a mutable owned or shared slice because heap blocks store their size.
public void append<type T>(&mut owned []T array, T value, ufloat growthRate=arrayGrowthRate) {
    // Reallocate `array` if the block it points to ran out of space.
    if ((*array).length + 1)*sizeof(T) > std.memory.blockSize(*array) {
        *array = new [array.length*arrayGrowthRate]T(*array);
    }
    std.memory.setLength(array, (*array).length + 1);
    (*array)[array.length] = value;
}
