pub package animals
from std.io import printLine

pub struct Animal {
    pub name string
    pub age uint32
}

pub method makeNoise(animal &Animal)

    embed Animal
    pub tag string
}

pub struct Cat {
    embed Animal
    pub breed string
}

pub func makeNoise(dog &Dog) {
    printLine("woof")
}

pub func makeNoise(cat &Cat) {
    printLine("meow")
}

pub func main() {
    var pet mut &Animal = &Dog()
    makeNoise(pet)
    pet = &Cat()
    pet->makeNoise()
}

pub package physics
import std.math
import std.operations

pub enum Entity {
    pub pos, vel, acc Vector
}

pub struct Circle {
    embed Entity
    pub radius float64
}

pub struct Mesh {
    embed Entity
    pub vertices owned[[]Vector]
}

pub struct Vector {
    from std.operations embed default Arithmetic, Equality
    pub x, y, z float64
}

pub func collide(a, b &mut Entity) {
    match a, b {
        case Circle, Circle:
        case Mesh, Mesh:
        case Circle, Mesh:
        case Mesh, Circle:
    }
}

pub func .length(v &Vector) {
    return std.math.sqrt(v.x*v.x + v.y*v.y)
}
