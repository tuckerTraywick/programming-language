pub package animals
from std.io import printLine

pub struct Animal {
    pub name string
    pub age uint
    pub makeNoise(&this)
}

pub struct Dog {
    impl Animal
    pub tag string
}

pub struct Cat {
    impl Animal
    pub breed string
}

pub func makeNoise(dog &Dog) {
    printLine("woof")
}

pub func makeNoise(cat &Cat) {
    printLine("meow")
}

pub func main() {
    var pet mut &Animal = &Dog()
    pet.makeNoise()
    pet = &Cat()
    pet.makeNoise()
}

pub package physics
import std.math
import std.traits

pub enum Entity {
    pub pos, vel, acc Vector
    pub collide(a, b &This)
}

pub struct Circle {
    impl Entity
    pub radius float
}

pub struct Mesh {
    impl Entity
    pub vertices owned[[]Vector]
}

pub struct Vector {
    from std.traits impl default Arithmetic, Equality
    pub x, y float
}

pub func collide(a, b &mut Entity) {
    match a, b {
        case Circle, Circle:
        case Mesh, Mesh:
        case Circle, Mesh:
        case Mesh, Circle:
    }
}

pub func .length(vector &Vector) float {
    return std.math.sqrt(vector.x*vector.x + vector.y*vector.y)
}
