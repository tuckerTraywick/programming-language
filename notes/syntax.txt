// Generic stuff above the definition leads to better readability.
generic[Element]
pub type List {
    length usize
    elements owned[[]Element]
}

generic[list]
pub fun .capacity(list &List) => auto


// This function is generic over `list`'s type parameters.
generic[list]
pub fun append(list &mut List, element &list.Element)


// You can add constraints to types.
generic[T] where T extends Add
fun add(a, b T) T {
    return a + b
}


// The constraints can span multiple lines
generic[T] where
    T extends ToString,
    T is not int,
pub fun printAnythingButInts(x T) {
    std.io.printLine(x->toString())
}

pub package animals
from std.io import printLine

pub type Animal {
    pub name string
    pub age uint32
    pub makeNoise(&this)
}

pub type Dog {
    extend Animal
    pub tag string
}

pub type Cat {
    extend Animal
    pub breed string
}

pub fun makeNoise(dog &Dog) {
    printLine("woof")
}

pub fun makeNoise(cat &Cat) {
    printLine("meow")
}

pub fun main() {
    var pet mut &Animal = &Dog()
    makeNoise(pet)
    pet = &Cat()
    pet->makeNoise()
}

pub package physics
from std.math import sqrt
from std.operations import *

pub type Entity {
    pub pos, vel, acc Vector
} cases {
    type Circle {
        pub radius float64
    }

    type Mesh {
        pub vertices []Vector
    }
}

pub type Vector {
    extend default Arithmetic, Equality
    pub x, y, z float64
}

pub fun collide(a, b &mut Entity) {
    match a, b {
        Circle, Circle:
        Mesh, Mesh:
        Circle, Mesh:
        Mesh, Circle:
    }
}

pub fun .length(v &Vector) {
    return sqrt(v.x*v.x + v.y*v.y)
}
