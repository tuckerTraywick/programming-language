- Keeping struct vs trait distinction because it lets you easily define a getter and setter at the
  same time and whether a type is abstract.
- Keeping tuple vs array vs map and Sum/Product vs struct/cases because it makes syntax neater.
- No embed vs impl because it makes things simpler and makes succinct definitions easier.

var tuple struct {_ int32, _ string} = {1, "abc"}
var array = []int32{1, 2, 3}
var array = [3]{} // An array of 3 elements.
var dict Map<string, int32> = {"a" = 1, "b" = 2}
var a = new [10] // An array of 10 elements.
var a = new {10} // An array of 1 element.

type UserId struct {uint32}
type IntOrString union {int32, string}
type File trait {Readable, Writeable}
type Direction struct {uint32} union {n, s, e, w}
type MyTuple struct {_ int32, _ string}
func returnMultipleValues() struct {a int32, b string}
func maybeInt() optional int32
func mayFail() union {int32, MyError}


var tuple (int32, string) = (1, "a")
var array [3]int32 = [1, 2, 3]
var dict <string>int32 = <"a" = 1, "b" = 2>
var a = new [] // an array of 0 elements
var a = new [10] // an array of 1 element
var a = new [] as [10]int32 // an array of 10 elements

struct MyAlias = int32 // Not a distinct type. Resolves to int32.
struct UserId {embed uint32} // A new distinct subtype of int32.
struct IntOrString {embed Sum<int32, string>}
trait File {embed Product<Readable, Writable>}
struct Direction {embed uint32} cases {n, s, e, w}
struct MyTuple {embed (int32, string)}
func returnMultipleValues() (a int32, b string)
func maybeInt() Optional<int32>
func mayFail() Sum<int32, MyError>


var heap owned &BlockAllocator = new BlockAllocator(1024)
var a owned &int32 = new 1 in heap
var b weak &int32 = a
drop heap

var stack owned &StackAllocator = new StackAllocator(1024)
var a owned &int32 = new 1 in stack
drop stack

// You can use the value of drop to move things.
var a owned &int32 = new 1
var b owned &int32 = drop a
// a is no longer valid and set to null so it will segfault if dereferenced.

func foo() {
	var x = new 1 // Uses implicit allocator parameter.
}

func main() {
	var a = new 1 // Uses global allocator.
	var heap = new BlockAllocator(1024)
	foo(allocator=heap) // Make foo use a different allocator.
}




pub module animals
from std.io import printLine

pub trait Animal {
	pub name string
	pub age uint32
	pub func makeNoise(&this)
}

pub struct Dog {
	embed Animal
	pub tag string
}

pub struct Cat {
	embed Animal
	pub breed string
}

pub func makeNoise(dog &Dog) {printLine("woof")}

pub func makeNoise(cat &Cat) {printLine("meow")}

func main() {
	var pet mut &Animal = &Dog()
	makeNoise(pet)
	pet = &Cat()
	pet->makeNoise()
}




pub module physics
import std.math
import std.operators

pub struct Entity {
	pub pos, vel, acc Vector
	pub func collide(a, b &mut This)
} cases {
	struct Circle {
		pub radius float64
	}

	struct Mesh {
		pub vertices owned []Vector
	}
}

pub struct Vector {
	embed std.operators.To<string>, .DefaultArithmetic<This>, .DefaultArithmetic<float64>
	pub x, y, z float64
}

pub func collide(a, b &mut Circle) {}

pub func collide(a, b &mut Mesh) {}

pub func collide(a &mut Circle, b &mut Mesh) {}

pub func collide(a &mut Mesh, b &mut Circle) {}

pub func to<string>(vector &Vector) {
	return f"({vector.x}, {vector.y})"
}




module color

	red = (255, 0, 0)

	blue = (0, 0, 255)

// You can only embed types with cases that are themselves types in a separate module to avoid name
// collision.
module cases
from color import *

	r uint8

	embed Color
	black = 0
}

module data
from color import *

	embed Color
	a uint8
} cases {
	red = (255, 0, 0, 255)
}




struct Shape {
	color string
	func area(&this) float64
} cases {
	struct Circle {r float64}
	struct Triangle {base, height float64}
	struct Rectangle {width, height float64}
}

func area(circle &Circle) => math.pi*circle.radius*circle.radius

func area(triangle &Triangle) => 0.5*triangle.base*triangle.height

func area(rectangle &Rectangle) => rectangle.width*rectangle.height

func main() {
	var shapes = [Circle(), Triangle(), Rectangle()]
	for shape in shapes {
		printLine(area(&shape))
	}
}




from std.arguments import *

func main(arguments []string) ExitCode {
	var help mut bool
	var number mut int32
	var x mut int32
	var numbers mut [3]int32

	namedArgument(&help, "help")
	namedArgument(&number, "number")
	positionalArgument(&x, required=true)
	namedArgument(&numbers, "numbers")
	try parseArguments() catch {
		printErrorMessage()
		return
	}

	if help {
		printHelpMessage()
	}
	return _.ok
}
