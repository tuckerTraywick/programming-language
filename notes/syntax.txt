pub module animals
from std.io import printLine

pub type Animal {
	pub name string
	pub age uint32
	pub func makeNoise(&this)
}

pub type Dog {
	embed Animal
	pub tag string
}

pub type Cat {
	embed Animal
	pub breed string
}

pub func makeNoise(dog &Dog) {printLine("woof")}

pub func makeNoise(cat &Cat) {printLine("meow")}

func main() {
	var pet mut &Animal = &Dog()
	makeNoise(pet)
	pet = &Cat()
	pet->makeNoise()
}

pub module physics
import std.math
import std.operators

pub type Entity {
	pub pos, vel, acc Vector
	pub func update(a, b &mut This)
} cases {
	type Circle {
		pub radius float64
	}

	type Mesh {
		pub vertices owned []Vector
	}
}

pub type Vector {
	embed std.operators.Equality<This>, std.operators.To<string>, .Arithmetic<This>, .Arithmetic<float64>
	x, y, z float64
}

pub func collide(a, b &mut Circle) {}

pub func collide(a, b &mut Mesh) {}

pub func collide(a &mut Circle, b &mut Mesh) {}

pub func collide(a &mut Mesh, b &mut Circle) {}

pub func to<string>(vector &Vector) {
	return f"({vector.x}, {vector.y})"
}
