pub package animals
from std.io import printLine

pub type Animal {
    pub name string
    pub age uint32
    pub makeNoise(&this)
}

pub type Dog {
    embed Animal
    pub tag string
}

pub type Cat {
    embed Animal
    pub breed string
}

pub fun makeNoise(dog &Dog) {
    printLine("woof")
}

pub fun makeNoise(cat &Cat) {
    printLine("meow")
}

pub fun main() {
    var pet mut &Animal = &Dog()
    pet.makeNoise()
    pet = &Cat()
    pet.makeNoise()
}


pub package moreAnimals
from std.io import printLine
from animals import *

pub type Pet {
    embed in Dog, Cat
    pub walk(&this)
}

pub type Hamster {
    embed Pet, Animal
}

pub fun walk(dog &Dog) {}

pub fun walk(cat &Cat) {}

pub fun walk(hamster &Hamster) {}

pub fun makeNoise(hamster &Hamster) {
    printLine("squeak")
}

    pub r, g, b uint8
} cases {
    default type Rgb {}
    red = (255, 0, 0)
    green = (0, 255, 0)
    blue = (0, 0, 255)
}

}
