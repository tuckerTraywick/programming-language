    struct Rectangle {
        ufloat64 width, height;
    }

        ufloat64 base, height;
    }

    struct Circle {
        ufloat64 radius;
    }
}

ufloat64 area(Shape shape);

ufloat64 area(Shape.Rectangle rectangle) {
    return rectangle.width*rectangle.height;
}

ufloat64 area(Shape.Triangle triangle) {
    return 0.5*triangle.base*triangle.height;
}

ufloat64 area(Shape.Circle circle) {
    return pi*pow(circle.radius, 2);
}

void main() {
    mut Shape shape = .Rectangle(2, 5);
    printLine(shape->area());
    shape = .Triangle(2, 5);
    printLine(shape->area());
}

public package std.flags;
import std.io.printLine;

void add() {
    setDescription("Adds 1 or 2 numbers.");
    setUsage("add [options] <number1> [number2]");
    
    mut bool help;
    mut int number1;
    mut optional int number2 = .Some(1); // default value
    addFlag(&help, {"h", "help"}, "Print this help message and exit.");
    addFlag(&number1, "number1", "The first number to add.");
    addFlag(&number2, "number2", "The second number to add. Default is 1.");
    optional FlagParsingError error = parseFlags(args);

    if error {
        printLine(error.message);
        return;
    }
}

void gcc() {
    mut []string files;
    mut []string outputPath;
    addArgs(&files, min=1);
    addFlag(&outputPath, "o", "The path of the output.", count=1); // trailing argument
    parseFlags();
    string path = outputPath[0];
}

void git() {
    mut bool help;
    addFlag(&help, {"h", "help"});

    mut bool clone;
    mut []string cloneArgs;
    mut optional int depth = 3;
    addCommand(&clone, "clone");
    addArgs(&cloneArgs, count=1);
    addFlag(&depth, "depth");
    
    mut bool add;
    mut []string addArgs;
    addCommand(&add, "add");
    addArgs(&addArgs, min=1);
}
