pub package animals
from std.io import printLine

pub type Animal {
    pub name string
    pub age uint
    pub makeNoise(animal &Animal)
}

pub type Dog {
    embed Animal
    pub tag string
}

pub type Cat {
    embed Animal
    pub breed string
}

pub fun makeNoise(dog &Dog) {
    printLine("woof")
}

pub fun makeNoise(cat &Cat) {
    printLine("meow")
}

pub fun main() {
    var pet mut &Animal = &Dog()
    pet.makeNoise() // prints "woof"
    pet = &Cat()
    pet.makeNoise() // prints "meow"
}

pub package physics
import std.math

pub type Entity {
    pub pos, vel, acc Vector
    pub collide(a, b &mut This)
} cases {
    type Circle {
        pub radius float64
    }

    type Mesh {
        pub vertices owned[[]Vector]
    }
}

pub type Vector {
    embed default std.traits.Arithmetic, .Equality
    pub x, y float64
}

pub fun collide(a, b &mut Entity) {
    match a, b {
        case Circle, Circle:
        case Mesh, Mesh:
        case Circle, Mesh:
        case Mesh, Circle:
    }
}

pub fun .length(vector &Vector) float64 {
    return std.math.sqrt(vector.x*vector.x + vector.y*vector.y)
}
