- 64-bit stack machine
- Load-store
- Operands always on the stack, besides the push instructions

Registers:
    ip: instruction pointer
    fp: frame pointer, points to the FIRST byte of the frame
    sp: stack pointer, points to the LAST byte of the stack

Instructions:
    opcode: 1 byte (aligned to 1 byte)
    value to push: up to 8 bytes (also aligned to 1 byte)

# No separate keyword for ementing a method because you will should be able to
# ement a new method for an already compiled struct.

# One ementation per method name per type, you cannot ement the same method differently
# for different interfaces.

# Is structural typing compatible with padding??
# If a struct has compatible cases, can it be cast???

# No structural typing, just "extend" and "in".

# "extend/ in" only works if the alignment is compatible.

# No postfix operators because they're too hard to distinguish from infix ones.

# Prefix operators are all higher precedence than infix operators (besides . [] ()).

# No right associative operators.

package shapes
from std.io import printLine
from std.math import pi, pow

struct Shape {
    color [3]uint8
    area(this)
} cases {
    struct Circle {
        radius ufloat64
    }

    struct Rectangle {
        width, height ufloat64
    }

    struct Triangle {
        base, height ufloat64
    }
}

func area(circle Shape.Circle) {
    return pi*pow(circle.radius, 2)
}

func area(rectangle Shape.Rectanlge) {
    return rectangle.width*rectangle.height
}

func area(triangle Shape.Triangle) {
    return 0.5*triangle.base*triangle.height
}

func main() {
    var shape mut Shape = .Rectangle(2, 5)
    printLine(shape.area()) # Prints "10".
    shape = .Triangle(2, 5)
    printLine(shape.area()) # Prints "5".
}

package add1
from std.io import printLine
from std.operations import to
from std.flags import *

func main() {
    setDescription("Adds one to a number.")
    setUsage("add1 [options] --number=[integer]")
    var help = addFlag[bool](["h", "help"], "Print this help message.")
    var number = addFlag[int]("number", "The number to add 1 to.")
    var path = addFlag[bool]("p", "the path of the file")
    var error = parseFlags()

    if error {
        printLine(error->to[string]())
        return
    }

    if help {
        printHelp()
        return
    }
    
    printLine(f"{number} + 1 = {number + 1}")
}
