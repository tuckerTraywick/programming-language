lineComment = "//" *!"\n"

newline = *"\n"

whitespace = [ \t\r]

identifier
    = [a-zA-Z_] *[a-zA-Z_0-9]

character
    = "'" (escape | !"'" | <Expected a character.>)
      ?(+!"'" <Invalid character.>)
      ("'" | <Missing closing '.>)

string
    = '"' *(escape | !'"')
      ('"' | <Missing closing ".>)

escape
    = "\\" ([nrt0'"\\] | <Invalid escape sequence.>)

number
    = +[0-9]


program
    = ?packageStatement *programStatement

packageStatement
    = accessModifier "package" dottedIdentifier

programStatement
    = importStatement
    | accessModifier "var" variableDefinition
    | accessModifier "fun" functionDefinition
    | accessModifier structDefinition

importStatement
    = "from" dottedIdentifier "import" dottedIdentifierList
    | "import" dottedIdentifier

dottedIdentifierList
    = "*"
    | *(dottedIdentifier ",") ?dottedIdentifier

structDefinition
    = "struct" identifier typeParameterList ?("{" *structMember "}" | "=" type)
      ?("cases" "{" *structCase "}")

structMember
    = accessModifier variableDefinition
    | accessModifier functionDefinition
    | "embed" typeList

structCase
    = identifierList ?("=" expressionList)
    | structDefinition
    | typeList
    | "..."

block
    = "{" *blockStatement "}"

blockStatement
    = "var" variableDefinition
    | "fun" functionDefinition
    | whileLoop
    | forLoop
    | ifStatement
    | switchStatement
    | returnStatement
    | continueStatement
    | breakStatement
    | fallthroughStatement
    | assignment
    | expression
    | block

variableDefinition
    = identifierList ("=" expressionList | type ?("=" expressionList))

functionDefinition
    = identifier typeParameterList functionParameterList ?("=>" expression | block)

whileLoop
    = ?"do" "while" expression block

forLoop
    = "for" forVariableList "in" expressionList block

forVariableList
    = *(forVariable ",") ?forVariable

forVariable
    = "(" forVariableList ")"
    | identifier ?type

ifStatement
    = "if" expression block *("else" "if" expression block) ?("else" block)

switchStatement
    = "switch" expressionList *("case" expressionList block) ?("default" *block)

returnStatement
    = "return" ?expression

continueStatement
    = "continue"

breakStatement
    = "break"

fallthroughStatement
    = "fallthrough"

assignment
    = expressionList assignmentOperator expressionList

type
    = arrayType
    | pointerType
    | tupleType
    | functionType
    | constType
    | mutType
    | ownedType
    | weakType
    | basicType
    | "auto"

arrayType
    = "[" ?expression "]" type

pointerType
    = "&" type

tupleType
    = "(" type *("," type) ")" *typeArgumentList

functionType
    = "fun" typeArgumentList functionParameterList ?type

constType
    = "const" type

mutType
    = "mut" type

ownedType
    = "owned" type

weakType
    = "weak" type

basicType
    = dottedIdentifier *typeArgumentList

typeParameterList
    = ?("[" *(typeParameter ",") ?typeParameter "]")

typeParameter
    = identifier ?(?"embeds" type)

functionParameterList
    = "(" *(functionParameter ",") ?functionParameter ")"

functionParameter
    = identifier ?type ?("=" expression)
    | type ?("=" expression)

identifierList
    = identifier *("," identifier)

expressionList
    = expression *("," expression)

expression
    = operation

basicExpression
    = arrayLiteral
    | (identifier | functionArgumentList) *(typeArgumentList | functionArgumentList | "." identifier)

arrayLiteral
    = "[" *(arrayElement ",") ?arrayElement "]"

arrayElement
    = expression ?("=" expression)

dottedIdentifier
    = identifier *("." identifier)

functionArgumentList
    = "(" *(functionArgument ",") ?functionArgument ")"

functionArgument
    = ?(identifier "=") expression

typeArgumentList
    = ?("[" *(typeArgument ",") ?typeArgument "]")

typeArgument
    = ?(identifier "=") (type | expression)

accessModifier
    = ?"pub"
