lineCommentStart = "//"

identifier
    = [a-zA-Z_] *[a-zA-Z_0-9]

character
    = "'" (!'\\' | '\\' number | '\\' _) "'"

string
    = '"' *(!'\\' | '\\' number | '\\' _) '"'

number
    = +[0-9]


program
    = ?packageStatement *programStatement

packageStatement
    = accessModifier "package" packageName

programStatement
    = importStatement
    | accessModifier variableDefinition
    | accessModifier functionDefinition
    | accessModifier structDefinition

importStatement
    = "from" packageName "import" packageNameList
    | "import" packageName

packageNameList
    = "*"
    | packageName *("," packageName) ?","

structDefinition
    = "struct" identifier typeParameterList ?("{" *structMember "}" | "=" type)
      ?("cases" "{" *structCase "}")

structMember
    = accessModifier variableDefinition
    | accessModifier functionDefinition
    | "embed" typeList

structCase
    = identifierList ?("=" expressionList)
    | structDefinition
    | "embed" typeList

block
    = "{" *blockStatement "}"

blockStatement
    = variableDefinition
    | functionDefinition
    | whileLoop
    | forLoop
    | ifStatement
    | switchStatement
    | returnStatement
    | continueStatement
    | breakStatement
    | fallthroughStatement
    | assignment
    | expression
    | block

variableDefinition
    = "var" identifierList ("=" expressionList | type ?("=" expressionList))

functionDefinition
    = "fun" identifier typeParameterList functionParameterList ?("=>" expression | block)

whileLoop
    = ?"do" "while" expression block

forLoop
    = "for" forVariableList "in" expressionList block

forVariableList
    = forVariable *("," forVariable) ?","

forVariable
    = "(" forVariableList ")"
    | identifier ?type

ifStatement
    = "if" expression block *("else" "if" expression block) ?("else" block)

switchStatement
    = "switch" expressionList *("case" expressionList block) ?("default" *block)

returnStatement
    = "return" ?expression

continueStatement
    = "continue"

breakStatement
    = "break"

fallthroughStatement
    = "fallthrough"

assignment
    = expressionList assignmentOperator expressionList

identifierList
    = identifier *("," identifier) ?","

expressionList
    = expression *("," expression) ?","

expression
    = range
    | number

range
    = expression (".." | "...") ?expression ?(":" expression)
    | (".." | "...") expression ?(":" expression)

typeParameterList
    = ?("[" *(typeParameter ",") ?typeParameter "]")

typeParameter
    = identifier ?(?"embeds" type)

functionParameterList
    = "(" *(functionParameter ",") ?functionParameter ")"

functionParameter
    = type
    | identifier type ?("=" expression)

type
    = packageName

packageName
    = identifier *("." identifier)

accessModifier
    = ?("public" | "private")
