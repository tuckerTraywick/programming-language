package fib
from std.types import 
from std.io import printLine

fun fib(n uint) uint {
    if n == 1 or n == 1 {
        return n
    }
    return fib(n - 1) + fib(n - 2)
}

fun main(args []string) {
    var number = to[uint](args[1])
    if number {
        printLine(fib())
    } else {
        print("Invalid number.")
    }
}
