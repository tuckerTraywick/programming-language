package fib
from std.types import to
from std.io import printLine

fun fib(n uint) uint {
    if n <= 1 {
        return n
    }
    return fib(n - 1) + fib(n - 2)
}

fun main(args []string) {
    var number = to[uint](args[1])
    if number {
        printLine("fib(" + number + ") =" + fib(number))
    } else {
        printLine("Invalid number.")
    }
}
