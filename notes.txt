struct Animal {
    name string
    age uint
    makeNoise()
}

struct Dog {
    embed Animal
    tag string
}

struct Cat {
    embed Animal
    breed string
}

func makeNoise(dog &Dog) {
    println("woof")
}

func makeNoise(cat &Cat) {
    println("meow")
}

func main() {
    var pet &Animal = &Dog()
    pet.makeNoise()
    pet = &Cat()
    pet.makeNoise()
}

# Assigning of vtables happens when the pointer is cast.
struct F {
    f()
}

struct A {
    F
    x int
}

struct B {
    F
    x int
}

func f(a &A) {
    printLine("A.f()")
}

func f(b &B) {
    printLine("B.f()")
}

var object &A = &B()
var trait &F = object
trait.f()
