def parseExpression():
    tree = None
    while tokens:
        token = peek()
        if token is prefix:
            addLeaf(tree, PrefixNode(consume(), [parseLiteral()]))
        elif token is infix:
            lhs = tree.innermost
            operator = self.consume()
            rhs = self.parseLiteral()
            if precedence(operator) > precedence(lhs):
                # Steal the innermost tree's last child.
                appendLeaf(lhs, InfixNode(operator, [lhs.children.pop(), rhs]))
            else:
                # Wrap the innermost tree.
                replaceLeaf(lhs, InfixNode(operator, [lhs, rhs]))
        elif tree is None:
            tree = parseLiteral()
        else:
            break

    if tokens:
        raise exception
    return tree

package add1
from std import *
from std.io import printLine

fun main() {
    description("Adds one to a number.")
    usage("add1 [options] --number=<integer>")
    parseFlags()
    
    var help = flag[bool](["h", "help"], "Print this help message.")
    var number = flag[int]("number=<integer>", "The number to add 1 to.")
    
    if help {
        printHelp()
    }
    
    if number {
        printLine(f"{number} + 1 = {number + 1}")
    }
}
