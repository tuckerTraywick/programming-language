# No separate keyword for implementing a method because you will should be able to
# implement a new method for an already compiled struct.

# One implementation per method name per type, you cannot implement the same method differently
# for different interfaces.

# Is structural typing compatible with padding??
# If a struct has compatible cases, can it be cast???

# No structural typing, just "extend" and "abstract".

# "abstract" only works if the alignment is compatible.

# No postfix operators because they're too hard to distinguish from infix ones.

def parseExpression():
    tree = None
    while tokens:
        token = peek()
        if token is prefix:
            addLeaf(tree, PrefixNode(consume(), [parseLiteral()]))
        elif token is infix:
            lhs = tree.innermost
            operator = self.consume()
            rhs = self.parseLiteral()
            if precedence(operator) > precedence(lhs):
                # Steal the innermost tree's last child.
                appendLeaf(lhs, InfixNode(operator, [lhs.children.pop(), rhs]))
            else:
                # Wrap the innermost tree.
                replaceLeaf(lhs, InfixNode(operator, [lhs, rhs]))
        elif tree is None:
            tree = parseLiteral()
        else:
            break

    if tokens:
        raise exception
    return tree

import std.io

    name string
    age uint
}

    age uint
    breed string
    name string
}

    abstract Dog, Cat
    &name() &uint = auto
    &age() &uint = auto
}

package compiler

struct Tree {
    top &Node
    errors ?[]&ParsingERror
}

struct Node {
    type NodeType
    parent ?&Node
    children ?[]&Leaf
}

struct NodeType cases {
    program, packageStatement, ...
}

struct &Leaf = &Token | &Node | &ParsingError

struct Token {
    type TokenType
    text string
    textIndex usize
}

struct TokenType cases {
    default invalid
    number, character, string, identifier
    PACKAGE, IMPORT, VAR, FUNC, PLUS, MINUS ...
}

struct ParsingError {
    message string
    tokenIndex usize
    textIndex usize
    row usize
    column usize
}

struct Parser {
    tokens []Token
    tokenIndex usize
    tree Tree
    bottomNode ?&Node # The innermost, rightmost node of the tree.
    bottomLeaf ?&Leaf # The rightmost leaf of the bottom node.
    keepParsing bool # Set to false when an unrecoverable error is encountered.
}

# Lexing functions
func lex(text string) []Token

# Parsing functions.
func parse(text string) Tree

func parse(tokens []Token) Tree {
    var parser = Parser(tokens=tokens, tree=&Node.program())
    parseProgram(&parser)
    return Tree(parser.topNode, parser.errors)
}

func .currentToken(parser &Parser) ?Token {
    if parser.tokenIndex < parser.tokens.length {
        return parser.tokens[parser.tokenIndex]
    }
    return none
}

func advance(parser &Parser) {
    assert parser.tokenIndex < parser.tokens.length, "advance() should only be called when tokens are left."
    ++parser.tokenIndex
}

func accept(parser &Parser, types ..TokenType, errorMessage ?string) {
    if parser.keepParsing and parser.currentToken and parser.currentToken.type in types {
        parser->advance()
    }
}

func expect(parser &Parser, types ..TokenType, errorMessage ?string) {
    if not parser.keepParsing {
        return
    }

    if parser.currentToken.type not in types {
        parser->addLeaf(&ParsingError(errorMessage))
    }
}

func maybe(parser &Parser, f func(&Parser))

func any(parser &Parser, fs ..func(&Parser))

func zeroOrMore(parser &Parser, f func(&Parser))

func oneOrMore(parser &Parser, f func(&Parser), errorMessage ?string)

func parseProgram(parser &Parser) {
    parsePackageStatement(parser)
}

func parsePackageStatement(parser &Parser) {
    parser->beginNode(.packageStatement)
        parser->accept(.PUB)
        parser->expect(.PACKAGE, "Expected a declaration.")
        parser->expect(.identifier, "Expected a package name.")
        parser->zeroOrMore(parser, func(parser &Parser) {
            parser->expect(.DOT)
            parser->expect(.identifier, .TIMES, "Incomplete package name.")
        })
    parser->endNode()
}
