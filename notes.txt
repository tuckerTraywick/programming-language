# No separate keyword for implementing a method because you will should be able to
# implement a new method for an already compiled struct.

# One implementation per method name per type, you cannot implement the same method differently
# for different interfaces.

# Is structural typing compatible with padding??
# If a struct has compatible cases, can it be cast???

# No structural typing, just "extend" and "abstract".

# "abstract" only works if the alignment is compatible.

# No postfix operators because they're too hard to distinguish from infix ones.

def parseExpression():
    tree = None
    while tokens:
        token = peek()
        if token is prefix:
            addLeaf(tree, PrefixNode(consume(), [parseLiteral()]))
        elif token is infix:
            lhs = tree.innermost
            operator = self.consume()
            rhs = self.parseLiteral()
            if precedence(operator) > precedence(lhs):
                # Steal the innermost tree's last child.
                appendLeaf(lhs, InfixNode(operator, [lhs.children.pop(), rhs]))
            else:
                # Wrap the innermost tree.
                replaceLeaf(lhs, InfixNode(operator, [lhs, rhs]))
        elif tree is None:
            tree = parseLiteral()
        else:
            break

    if tokens:
        raise exception
    return tree

package abstractingExample
import std.io

struct Dog {
    name string
    age uint
}

struct Cat {
    age uint
    breed string
    name string
}

struct Animal {
    abstract Dog, Cat
    &name() &uint = auto
    &age() &uint = auto
}

func printNameAndAge(pet &Animal) {
    std.io.printLine(f"name={pet.name}, age={pet.age}")
}

    printNameAndAge(&Dog("fido", 1))
    printNameAndAge(&Cat(1, "tabby", "felix"))

package compiler

struct CharIndex = usize

struct NodeIndex = usize

struct Node {
    variant uint16 = case
    variantName string = case
    textStart CharIndex
    textEnd CharIndex
    parent NodeIndex
    lastChild NodeIndex
    next NodeIndex
} cases {
    default invalid
    number, character, string, identifier
    PACKAGE, IMPORT, VAR, FUNC, PLUS, MINUS ...
    program, packageStatement, ...
}

struct Tree {
    embed std.traits.Pretty
    text string
    nodes []Node
    topNode NodeIndex
}

struct ParsingError {
    message string
    index usize
}

struct ParsingResult = Result[Tree, ParsingError]

struct Parser {
    tree &Tree
    error ?ParsingError
    currentToken usize
    topNode NodeIndex
    bottomNode NodeIndex
}

# Tree functions
func getNode(tree &Tree, index NodeIndex) &Node {
    assert index < tree.nodes.length, "Node index out of bounds."
    return tree.nodes[index]
}

func setNode(tree &Tree, index NodeIndex, value &Node) &Node {
    assert index < tree.nodes.length, "Node index out of bounds."
    tree.nodes[index] = value
}

func firstChild(tree &Tree, index NodeIndex) NodeIndex {
    var lastChildIndex = tree->getNode(index).lastChild
    return tree->getNode(lastChildIndex).next
}

func text(tree &tree, index NodeIdnex) string {
    var node = tree->getNode(index)
    return tree.text[node.textStart thru node.textEnd]
}

# Lexing functions
func lex(text string) ParsingResult

# Parsing functions
func zeroOrMore(parser &Parser, f func(&Parser)) {
    if parser.error {
        return
    }

    var previousState = *parser
    do {
        f(parser)
    } while not parser.error and parser.currentToken < parser.tree.nodes.length

    if parser.error {
        *parser = previousState
    }
}

func parse(tree &Tree) ParsingResult {
    var parser = Parser(tree=tree)
    return parseProgram(&parser)
}

func parsePackageStatement(parser &Parser) {
    parser->beginNode(.PACKAGE)
    parser->expect(.identifier)
    zeroOrMore(parser, func(parser) {
        parser->expect(.DOT)
        parser->expect(.identifier)
    })
    parser->endNode()
}
