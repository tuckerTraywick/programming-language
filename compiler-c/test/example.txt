public struct Animal {
    public string name;
    public uint age;
    public void makeNoise(*this);
}

public struct Dog {
    embed Animal;
    public string tag;
}

public struct Cat {
    embed Animal;
    public string breed;
}

public void makeNoise(*Dog dog) {
    printLine("woof");
}

public void makeNoise(*Cat cat) {
    printLine("meow");
}

void main() {
    mut *Animal pet = &Dog();
    pet.makeNoise();
    pet = &Cat();
    pet.makeNoise();
}
