pub package animals
from std.io import printLine

pub struct Animal {
    pub var name string
    pub var age uint
    pub fun makeNoise(&this)
}

pub struct Dog {
    embed Animal
    pub var tag string
}

pub struct Cat {
    embed Animal
    pub var breed string
}

pub fun makeNoise(dog &Dog) {
    printLine("woof")
}

pub fun makeNoise(cat &Cat) {
    printLine("meow")
}

pub package physics
import std.operations

pub struct Entity {
    pub var position, velocity acceleration Vector
    pub fun update(&mut this, delta ufloat64)
} cases {
    struct Circle {
        pub var radius ufloat64
    }

    struct Mesh {
        pub var vertices owned []Vector
    }
}

pub struct Vector {
    from std.operations embed Arithmetic, Equality, To[string]
    pub var x, y, z float64
}

pub fun update(entity &mut Entity.Circle, delta) {}

pub fun update(entity &mut Entity.Mesh, delta) {}

pub package std.collections

pub struct List[T] {
    var first mut optional owned &Node[T]
    var last mut optional &Node[T]
    var length mut usize
}

struct Node[T] {
    var next mut optional owned &Node[T]
    var previous mut optional &Node[T]
    var value mut T
}
